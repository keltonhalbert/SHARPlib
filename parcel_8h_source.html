299 template <typename Lft>
300 void lift_parcel(Lft& liftpcl, const float pressure_arr[],
301 float pcl_vtmpk_arr[], const std::ptrdiff_t N) {
302 // Lift the parcel from the LPL to the LCL
303 float pres_lcl;
304 float tmpk_lcl;
305 drylift(this->pres, this->tmpk, this->dwpk, pres_lcl, tmpk_lcl);
306 // If we are lifting elevated parcel (i.e. EIL), we need to make
307 // sure our LCL isnt above the top of our data.
308 if (pres_lcl < pressure_arr[N - 1]) return;
309
310 this->lcl_pressure = pres_lcl;
311
312 const float qv_lcl = mixratio(pres_lcl, tmpk_lcl);
313 const float thetav_lcl =
314 theta(pres_lcl, virtual_temperature(tmpk_lcl, qv_lcl),
315 THETA_REF_PRESSURE);
316
317 // Define the dry and saturated lift layers
318 PressureLayer dry_lyr = {this->pres, this->lcl_pressure};
319 PressureLayer sat_lyr = {this->lcl_pressure, pressure_arr[N - 1]};
320
321 // The LayerIndex excludes the top and bottom for interpolation reasons
322 const LayerIndex dry_idx = get_layer_index(dry_lyr, pressure_arr, N);
323 const LayerIndex sat_idx = get_layer_index(sat_lyr, pressure_arr, N);
324
325 for (std::ptrdiff_t k = 0; k < dry_idx.kbot; ++k) {
326 pcl_vtmpk_arr[k] = MISSING;
327 }
328
329 // Virtual potential temperature (Theta-V)
330 // is conserved for a parcels dry ascent to the LCL
331 for (std::ptrdiff_t k = dry_idx.kbot; k < dry_idx.ktop + 1; ++k) {
332 const float pcl_vtmp =
333 theta(THETA_REF_PRESSURE, thetav_lcl, pressure_arr[k]);
334 pcl_vtmpk_arr[k] = pcl_vtmp;
335 }
336
337 float pres_bot = pres_lcl;
338 float tmpk_bot = tmpk_lcl;
339 liftpcl.setup(pres_lcl, tmpk_lcl);
340
341 // fill the array with the moist parcel buoyancy
342 for (std::ptrdiff_t k = sat_idx.kbot; k < N; ++k) {
343 // compute above-lcl buoyancy here
344 const float pcl_pres = pressure_arr[k];
345 const float pcl_tmpk = liftpcl(pres_bot, tmpk_bot, pcl_pres);
346
347 if constexpr (!Lft::lift_from_lcl) {
348 pres_bot = pcl_pres;
349 tmpk_bot = pcl_tmpk;
350 }
351
352 const float pcl_vtmpk =
353 liftpcl.parcel_virtual_temperature(pcl_pres, pcl_tmpk);
354 pcl_vtmpk_arr[k] = pcl_vtmpk;
355 }
356 }

435 template <typename Lyr>
436 static Parcel mixed_layer_parcel(Lyr& mix_layer, const float pressure[],
437 [[maybe_unused]] const float height[],
438 const float pot_temperature[],
439 const float wv_mixratio[],
440 const std::ptrdiff_t N) noexcept {
441 float mean_mixr, mean_thta, pcl_pres;
442 if constexpr (Lyr::coord == LayerCoordinate::pressure) {
443 mean_mixr = layer_mean(mix_layer, pressure, wv_mixratio, N);
444 mean_thta = layer_mean(mix_layer, pressure, pot_temperature, N);
445 pcl_pres = mix_layer.bottom;
446
447 } else {
448 mean_mixr = layer_mean(mix_layer, height, pressure, wv_mixratio, N);
449 mean_thta =
450 layer_mean(mix_layer, height, pressure, pot_temperature, N);
451 pcl_pres =
452 sharp::interp_height(mix_layer.bottom, height, pressure, N);
453 }
454 const float tmpk = theta(THETA_REF_PRESSURE, mean_thta, pcl_pres);
455 const float dwpk = temperature_at_mixratio(mean_mixr, pcl_pres);
456
457 return Parcel(pcl_pres, tmpk, dwpk, LPL::ML);
458 }

483 template <typename Lyr, typename Lft>
484 static Parcel most_unstable_parcel(
485 Lyr& search_layer, Lft& lifter, const float pressure[],
486 const float height[], const float temperature[], const float virtemp[],
487 const float dewpoint[], float pcl_virtemp[], float buoy_arr[],
488 const std::ptrdiff_t N) noexcept {
489 LayerIndex lyr_idx;
490 if constexpr (Lyr::coord == LayerCoordinate::pressure) {
491 lyr_idx = get_layer_index(search_layer, pressure, N);
492
493 } else {
494 lyr_idx = get_layer_index(search_layer, height, N);
495 }
496
497 Parcel max_parcel;
498 for (std::ptrdiff_t idx = lyr_idx.kbot; idx <= lyr_idx.ktop; ++idx) {
499 const float pres = pressure[idx];
500 const float tmpk = temperature[idx];
501 const float dwpk = dewpoint[idx];
502 Parcel pcl(pres, tmpk, dwpk, LPL::MU);
503
504 pcl.lift_parcel(lifter, pressure, pcl_virtemp, N);
505 buoyancy(pcl_virtemp, virtemp, buoy_arr, N);
506 pcl.cape_cinh(pressure, height, buoy_arr, N);
507 if (pcl.cape > max_parcel.cape) max_parcel = pcl;
508 }
509
510 return max_parcel;
511 }
