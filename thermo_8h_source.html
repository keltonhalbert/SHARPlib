436[[nodiscard]] float wetbulb(Lft lifter, float pressure, float temperature,
437 float dewpoint) {
438#ifndef NO_QC
439 if ((pressure == MISSING) || (temperature == MISSING) ||

451 lifter.setup(pres_lcl, tmpk_lcl);
452

454}
