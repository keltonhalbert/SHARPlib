437[[nodiscard]] float wetbulb(Lft lifter, float pressure, float temperature,
438 float dewpoint) {
439#ifndef NO_QC
440 if ((pressure == MISSING) || (temperature == MISSING) ||

451 drylift(pressure, temperature, dewpoint, pres_lcl, tmpk_lcl);
452 lifter.setup(pres_lcl, tmpk_lcl);
453

455}
