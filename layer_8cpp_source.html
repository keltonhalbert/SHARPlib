29HeightLayer::HeightLayer(float bottom, float top, float delta) {
30 if ((std::isnan(bottom)) || (std::isnan(top))) {
31 throw std::range_error(fmt::format(
32 "RangeError: NaNs are not a valid range option for a HeightLayer. "
33 "Got bottom: {0} and top: {1}",
34 bottom, top));
35 }
36 if ((std::isinf(bottom)) || (std::isinf(top))) {
37 throw std::range_error(fmt::format(
38 "RangeError: infs are not a valid range option for a HeightLayer. "
39 "Got bottom: {0} and top: {1}",
40 bottom, top));
41 }
42 if (bottom > top) {
43 throw std::range_error(fmt::format(
44 "RangeError: The top of the height layer must be > the bottom of "
45 "the height layer. Got hbot: {0} and htop: {1}",
46 bottom, top));
47 }
48 this->bottom = bottom;
49 this->top = top;
50 this->delta = delta;
51}

93PressureLayer height_layer_to_pressure(HeightLayer layer,
94 const float pressure[],
95 const float height[],
96 const std::ptrdiff_t N,
97 const bool isAGL) {
98 if (isAGL) {
99 layer.bottom += height[0];
100 layer.top += height[0];
101 }
102
103 if ((layer.bottom < height[0]) || (layer.top > height[N - 1])) {
104 return {MISSING, MISSING};
105 }
106
107 const float pbot = interp_height(layer.bottom, height, pressure, N);
108 const float ptop = interp_height(layer.top, height, pressure, N);
109
110 return {pbot, ptop};
111}

113HeightLayer pressure_layer_to_height(PressureLayer layer,
114 const float pressure[],
115 const float height[],
116 const std::ptrdiff_t N, const bool toAGL) {
117 if ((layer.bottom > pressure[0]) || (layer.top < pressure[N - 1])) {
118 return {MISSING, MISSING};
119 }
120
121 float zbot = interp_pressure(layer.bottom, pressure, height, N);
122 float ztop = interp_pressure(layer.top, pressure, height, N);
123
124 if (toAGL) {
125 zbot -= height[0];
126 ztop -= height[0];
127 }
128
129 return {zbot, ztop};
130}
