39WindComponents storm_motion_bunkers(
40 const float pressure[], const float height[], const float u_wind[],
41 const float v_wind[], const std::ptrdiff_t N,
42 HeightLayer mean_wind_layer_agl, HeightLayer wind_shear_layer_agl,
43 const bool leftMover, const bool pressureWeighted) {
44 constexpr float deviation = 7.5; // deviation from mean wind in m/s
45
46 PressureLayer mw_lyr = height_layer_to_pressure(mean_wind_layer_agl,
47 pressure, height, N, true);
48
49 WindComponents layer_mean_wind = {MISSING, MISSING};
50 layer_mean_wind =
51 mean_wind(mw_lyr, pressure, u_wind, v_wind, N, pressureWeighted);
52
53 // The shear is computed by finding the 500m deep
54 // mean winds at the top and bottom of the wind_shear_layer
55 // and then differencing the two. Means are not pressure weighted
56 // for the non-parcel based method.
57 HeightLayer h_layer_lo = {wind_shear_layer_agl.bottom,
58 wind_shear_layer_agl.bottom + 500.0f};
59 HeightLayer h_layer_hi = {wind_shear_layer_agl.top - 500.0f,
60 wind_shear_layer_agl.top};
61
62 PressureLayer p_layer_lo =
63 height_layer_to_pressure(h_layer_lo, pressure, height, N, true);
64 PressureLayer p_layer_hi =
65 height_layer_to_pressure(h_layer_hi, pressure, height, N, true);
66
67 WindComponents winds_lo =
68 mean_wind(p_layer_lo, pressure, u_wind, v_wind, N, false);
69 WindComponents winds_hi =
70 mean_wind(p_layer_hi, pressure, u_wind, v_wind, N, false);
71
72 const float shear_u = winds_hi.u - winds_lo.u;
73 const float shear_v = winds_hi.v - winds_lo.v;
74 const float mag = vector_magnitude(shear_u, shear_v);
75
76 float storm_u = MISSING;
77 float storm_v = MISSING;
78
79 if (leftMover) {
80 storm_u = layer_mean_wind.u - ((deviation / mag) * shear_v);
81 storm_v = layer_mean_wind.v + ((deviation / mag) * shear_u);
82 } else {
83 storm_u = layer_mean_wind.u + ((deviation / mag) * shear_v);
84 storm_v = layer_mean_wind.v - ((deviation / mag) * shear_u);
85 }
86 return {storm_u, storm_v};
87}

89[[nodiscard]] WindComponents storm_motion_bunkers(
90 const float pressure[], const float height[], const float u_wind[],
91 const float v_wind[], const std::ptrdiff_t N, PressureLayer eff_infl_lyr,
92 const Parcel &mupcl, const bool leftMover) {
93 HeightLayer shr_layer = {0, 6000.0};
94 HeightLayer dflt_mw_lyr = {0.0, 6000.0};
95
96 if (mupcl.eql_pressure == MISSING) {
97 return storm_motion_bunkers(pressure, height, u_wind, v_wind, N,
98 dflt_mw_lyr, shr_layer, leftMover, false);
99 }
100
101 const float eql_pres = mupcl.eql_pressure;
102 if ((eff_infl_lyr.bottom == MISSING) || (eff_infl_lyr.top == MISSING)) {
103 return storm_motion_bunkers(pressure, height, u_wind, v_wind, N,
104 dflt_mw_lyr, shr_layer, leftMover, false);
105 }
106
107 HeightLayer eil_hght =
108 pressure_layer_to_height(eff_infl_lyr, pressure, height, N, true);
109
110 float eql_ht = interp_pressure(eql_pres, pressure, height, N);
111 // get AGL
112 eql_ht -= height[0];
113 const float htop = 0.65 * (eql_ht - eil_hght.bottom);
114
115 if ((htop < 3000.0f) || (eil_hght.bottom > htop)) {
116 return storm_motion_bunkers(pressure, height, u_wind, v_wind, N,
117 dflt_mw_lyr, shr_layer, leftMover, false);
118 }
119
120 HeightLayer mw_layer = {eil_hght.bottom, htop};
121 return storm_motion_bunkers(pressure, height, u_wind, v_wind, N, mw_layer,
122 shr_layer, leftMover, true);
123}

125float entrainment_cape(const float pressure[], const float height[],
126 const float temperature[], const float mse_arr[],
127 const float u_wind[], const float v_wind[],
128 const std::ptrdiff_t N, Parcel *pcl) {
129 // if cape is zero, we get a divide by zero issue later.
130 // there can "technically" be LFC/EL without CAPE because of very,
131 // very shallow buoyancy near zero when searching for LFC/EL.
132 // To-Do: maybe refactor LFC/EL search?
133 if ((pcl->lfc_pressure == MISSING) || (pcl->lfc_pressure == MISSING) ||
134 (pcl->cape == 0)) {
135 return 0.0;
136 }
137
138 float *mse_diff = new float[N];
139 float *mse_bar = new float[N];
140
141 // compute MSE_bar
142 mse_bar[0] = mse_arr[0];
143 const float psfc = pressure[0];
144 for (std::ptrdiff_t k = 1; k < N; ++k) {
145 PressureLayer mn_lyr = {psfc, pressure[k]};
146 mse_bar[k] = layer_mean(mn_lyr, pressure, mse_arr, N);
147 }
148
149 // compute MSE_star
150 const float hsfc = height[0];
151 for (std::ptrdiff_t k = 0; k < N; ++k) {
152 const float tmpk = temperature[k];
153 const float rsat = mixratio(pressure[k], tmpk);
154 const float qsat = specific_humidity(rsat);
155 const float height_agl = height[k] - hsfc;
156 const float mse_star = moist_static_energy(height_agl, tmpk, qsat);
157 mse_diff[k] = buoyancy_dilution_potential(tmpk, mse_bar[k], mse_star);
158 }
159
160 // compute NCAPE
161 PressureLayer plyr = {pcl->lfc_pressure, pcl->eql_pressure};
162 HeightLayer hlyr = pressure_layer_to_height(plyr, pressure, height, N);
163 const float NCAPE = integrate_layer_trapz(hlyr, mse_diff, height, N);
164
165 // Compute the Bunkers non-parcel based storm motion
166 WindComponents strm_mtn =
167 storm_motion_bunkers(pressure, height, u_wind, v_wind, N, {0.0, 6000.0},
168 {0.0, 6000.0}, false, false);
169
170 // get the mean 0-1km storm relative wind
171 HeightLayer layer = {hsfc + 0.0f, hsfc + 1000.0f};
172 LayerIndex layer_idx = get_layer_index(layer, height, N);
173
174 // loop from the surface to the last level before 1km AGL.
175 float V_sr_mean = 0.0;
176 int count = 0;
177 for (std::ptrdiff_t k = 0; k < layer_idx.ktop + 1; ++k) {
178#ifndef NO_QC
179 if ((u_wind[k] == MISSING) || (v_wind[k] == MISSING)) {
180 continue;
181 }
182#endif
183 V_sr_mean +=
184 vector_magnitude(u_wind[k] - strm_mtn.u, v_wind[k] - strm_mtn.v);
185 count += 1;
186 }
187 const float u_1km = interp_height(layer.top, height, u_wind, N);
188 const float v_1km = interp_height(layer.top, height, v_wind, N);
189
190 V_sr_mean += vector_magnitude(u_1km - strm_mtn.u, v_1km - strm_mtn.v);
191 V_sr_mean = V_sr_mean / (count + 1);
192
193 // now for all of the ECAPE nonsense
194 constexpr float L = 120.0;
195 const float H =
196 interp_pressure(pcl->eql_pressure, pressure, height, N) - hsfc;
197 constexpr float sigma = 1.6;
198 constexpr float alpha = 0.8;
199 const float pitchfork = (VKSQ * (alpha * alpha) * (PI * PI) * L) /
200 (PRANDTL * (sigma * sigma) * H);
201 const float V_sr_tilde = V_sr_mean / std::sqrt(2.0f * pcl->cape);
202 const float V_sr_tilde_sq = V_sr_tilde * V_sr_tilde;
203 const float N_tilde = NCAPE / pcl->cape;
204
205 const float term1 = pitchfork / V_sr_tilde_sq;
206 const float term2 = 1.0f + pitchfork + term1 * N_tilde;
207 const float term3 = 4.0f * term1 * (1.0f - pitchfork * N_tilde);
208 const float sqrt_term = term2 * term2 + term3;
209
210 // in the case of a negative solution,
211 // set ECAPE to 0;
212 if (sqrt_term < 0) {
213 return 0;
214 }
215
216 const float E_tilde = V_sr_tilde_sq + (-1.0f - pitchfork - (term1)*N_tilde +
217 std::sqrt(sqrt_term)) /
218 (2.0f * term1);
219 delete[] mse_diff;
220 delete[] mse_bar;
221
222 return E_tilde * pcl->cape;
223}

234float supercell_composite_parameter(float mu_cape, float eff_srh,
235 float eff_shear) {
236#ifndef NO_QC
237 if ((mu_cape == MISSING) || (eff_srh == MISSING) ||
238 (eff_shear == MISSING)) {
239 return MISSING;
240 }
241#endif
242
243 if (eff_shear > 20.0) {
244 eff_shear = 20.0;
245 } else if (eff_shear < 10.0) {
246 eff_shear = 0.0;
247 }
248
249 const float mu_cape_term = mu_cape / 1000.0f;
250 const float eff_srh_term = eff_srh / 50.0f;
251 const float eff_shear_term = eff_shear / 20.0f;
252
253 return mu_cape_term * eff_srh_term * eff_shear_term;
254}

256float significant_tornado_parameter(Parcel pcl, float lcl_hght_agl,
257 float storm_relative_helicity,
258 float bulk_wind_difference) {
259 float cinh_term, lcl_term, shear_term, srh_term, cape_term;
260 if (pcl.cape == MISSING) return MISSING;
261
262 if (pcl.cinh >= -50.0)
263 cinh_term = 1.0;
264 else if (pcl.cinh < -200.0)
265 cinh_term = 0.0;
266 else
267 cinh_term = ((200.0 + pcl.cinh) / 150.0);
268
269 // units of comparisons are meters
270 if (lcl_hght_agl < 1000.0)
271 lcl_term = 1.0;
272 else if (lcl_hght_agl > 2000.0)
273 lcl_term = 0.0;
274 else
275 lcl_term = ((2000.0 - lcl_hght_agl) / 1000.0);
276
277 // units of comparisons are m/s
278 if (bulk_wind_difference > 30)
279 shear_term = 1.5;
280 else if (bulk_wind_difference < 12.5)
281 shear_term = 0.0;
282 else
283 shear_term = bulk_wind_difference / 20.0;
284
285 if (storm_relative_helicity == MISSING)
286 srh_term = 0.0;
287 else
288 srh_term = storm_relative_helicity / 150.0;
289
290 cape_term = pcl.cape / 1500.0;
291
292 float stp = cape_term * cinh_term * lcl_term * srh_term * shear_term;
293 if (stp < 0) stp = 0;
294 return stp;
295}

297float significant_hail_parameter(const sharp::Parcel &mu_pcl,
298 float lapse_rate_700_500mb, float tmpk_500mb,
299 float freezing_lvl_agl, float shear_0_6km) {
300 float mu_mixr =
301 mixratio(mu_pcl.pres, mu_pcl.dwpk) * 1000.0f; // convert to g/kg
302
303 // restrict the inputs to specific ranges...
304 shear_0_6km = std::min(shear_0_6km, 27.0f);
305 shear_0_6km = std::max(shear_0_6km, 7.0f);
306
307 mu_mixr = std::min(mu_mixr, 13.6f);
308 mu_mixr = std::max(mu_mixr, 11.0f);
309
310 tmpk_500mb = std::min(tmpk_500mb, -5.5f + ZEROCNK);
311
312 float ship = (mu_pcl.cape * mu_mixr * lapse_rate_700_500mb *
313 ((tmpk_500mb - ZEROCNK) * -1) * shear_0_6km) /
314 42000000.0f;
315
316 if (mu_pcl.cape < 1300.0f) {
317 ship = ship * (mu_pcl.cape / 1300.0f);
318 }
319
320 if (lapse_rate_700_500mb < 5.8f) {
321 ship = ship * (lapse_rate_700_500mb / 5.8f);
322 }
323
324 if (freezing_lvl_agl < 2400.0f) {
325 ship = ship * (freezing_lvl_agl / 2400.0f);
326 }
327
328 return ship;
329}
