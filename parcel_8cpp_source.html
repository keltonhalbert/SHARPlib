76void Parcel::find_lfc_el(const float pres_arr[], const float hght_arr[],
77 const float buoy_arr[], const std::ptrdiff_t N) {
78 PressureLayer sat_lyr = {this->lcl_pressure, pres_arr[N - 1]};
79 LayerIndex lyr_idx = get_layer_index(sat_lyr, pres_arr, N);
80
81 float lyr_bot = 0.0;
82 float pos_buoy = 0.0;
83 float pos_buoy_last = 0.0;
84 float pbot = sat_lyr.bottom;
85 float buoy_bot = interp_pressure(sat_lyr.bottom, pres_arr, buoy_arr, N);
86 float hbot = interp_pressure(sat_lyr.bottom, pres_arr, hght_arr, N);
87 // set the LFC pressure to the LCL if the buoyancy is positive
88 float lfc_pres = (buoy_bot > 0) ? sat_lyr.bottom : MISSING;
89 float eql_pres = MISSING;
90 float lfc_pres_last = MISSING;
91 float eql_pres_last = MISSING;
92
93 for (std::ptrdiff_t k = lyr_idx.kbot; k < lyr_idx.ktop + 1; ++k) {
94#ifndef NO_QC
95 if (buoy_arr[k] == MISSING) continue;
96#endif
97 const float ptop = pres_arr[k];
98 const float htop = hght_arr[k];
99 const float buoy_top = buoy_arr[k];
100 const float lyr_top = (buoy_top + buoy_bot) / 2.0f;
101 // LFC condition
102 if ((lyr_bot <= 0) && (lyr_top > 0)) {
103 if (lfc_pres != MISSING) {
104 pos_buoy_last = pos_buoy;
105 lfc_pres_last = lfc_pres;
106 eql_pres_last = eql_pres;
107 pos_buoy = 0.0;
108 }
109 for (lfc_pres = pbot - 500; lfc_pres > ptop + 500;
110 lfc_pres -= 100.0) {
111 const float buoy =
112 interp_pressure(lfc_pres, pres_arr, buoy_arr, N);
113 if (buoy > 0) break;
114 }
115 }
116
117 // keep track of buoyancy so that we pick the max CAPE layer
118 const float condition = ((lfc_pres != MISSING) & (lyr_top > 0));
119 pos_buoy += condition * lyr_top * (htop - hbot);
120 // EL condition
121 if ((lfc_pres != MISSING) && ((lyr_bot >= 0) && (lyr_top < 0))) {
122 for (eql_pres = pbot - 500; eql_pres > ptop + 500;
123 eql_pres -= 100.0) {
124 const float buoy =
125 interp_pressure(eql_pres, pres_arr, buoy_arr, N);
126 if (buoy < 0) break;
127 }
128 if (pos_buoy_last > pos_buoy) {
129 lfc_pres = lfc_pres_last;
130 eql_pres = eql_pres_last;
131 pos_buoy = pos_buoy_last;
132 }
133 }
134 // If there is no EL, just use the last available level
135 if ((k == N - 1) && (lyr_top > 0)) eql_pres = pres_arr[N - 1];
136 // set loop variables
137 pbot = ptop;
138 hbot = htop;
139 buoy_bot = buoy_top;
140 lyr_bot = lyr_top;
141 }
142 if (pos_buoy > 0.0f) {
143 this->lfc_pressure = lfc_pres;
144 this->eql_pressure = eql_pres;
145 }
146}

148void Parcel::cape_cinh(const float pres_arr[], const float hght_arr[],
149 const float buoy_arr[], const ptrdiff_t N) {
150 if (this->lcl_pressure == MISSING) return;
151 find_lfc_el(pres_arr, hght_arr, buoy_arr, N);
152 if ((this->lfc_pressure != MISSING) && (this->eql_pressure != MISSING)) {
153 PressureLayer lfc_el = {this->lfc_pressure, this->eql_pressure};
154 PressureLayer lpl_lfc = {this->pres, this->lfc_pressure};
155 HeightLayer lfc_el_ht =
156 pressure_layer_to_height(lfc_el, pres_arr, hght_arr, N);
157 HeightLayer lpl_lfc_ht =
158 pressure_layer_to_height(lpl_lfc, pres_arr, hght_arr, N);
159
160 this->cinh =
161 integrate_layer_trapz(lpl_lfc_ht, buoy_arr, hght_arr, N, -1);
162 this->cape = integrate_layer_trapz(lfc_el_ht, buoy_arr, hght_arr, N, 1);
163 }
