76void Parcel::find_lfc_el(const float pres_arr[], const float hght_arr[],
77 const float buoy_arr[], const std::ptrdiff_t N) {
78 if (this->lcl_pressure <= pres_arr[N - 1]) return;
79 PressureLayer sat_lyr = {this->lcl_pressure, pres_arr[N - 1]};
80 LayerIndex lyr_idx = get_layer_index(sat_lyr, pres_arr, N);
81
82 float pos_buoy = 0.0;
83 float pos_buoy_max = 0.0;
84 float pbot = sat_lyr.bottom;
85 float buoy_bot = interp_pressure(sat_lyr.bottom, pres_arr, buoy_arr, N);
86 float hbot = interp_pressure(sat_lyr.bottom, pres_arr, hght_arr, N);
87 // set the LFC pressure to the LCL if the buoyancy is positive
88 float lfc_pres = (buoy_bot > 0) ? sat_lyr.bottom : MISSING;
89 float eql_pres = MISSING;
90 float lfc_pres_final = MISSING;
91 float eql_pres_final = MISSING;
92 bool in_pos_area = (buoy_bot > 0);
93
94 for (std::ptrdiff_t k = lyr_idx.kbot; k < lyr_idx.ktop + 1; ++k) {
95#ifndef NO_QC
96 if (buoy_arr[k] == MISSING) continue;
97#endif
98 const float ptop = pres_arr[k];
99 const float htop = hght_arr[k];
100 const float buoy_top = buoy_arr[k];
101 const float lyr_top = (buoy_top + buoy_bot) / 2.0f;
102
103 if (!in_pos_area && (buoy_bot <= 0) && (buoy_top > 0)) {
104 pos_buoy = 0.0;
105 eql_pres = MISSING;
106 in_pos_area = true;
107
108 const float log_pbot = std::log10(pbot);
109 const float log_ptop = std::log10(ptop);
110 const float buoy_diff = buoy_top - buoy_bot;
111 const float log_pres_diff = log_ptop - log_pbot;
112 const float log_lfc_pres =
113 log_pbot - buoy_bot * (log_pres_diff / buoy_diff);
114 lfc_pres =
115 std::min(std::pow(10.0f, log_lfc_pres), this->lcl_pressure);
116 }
117
118 const float condition = ((in_pos_area) & (lyr_top > 0));
119 pos_buoy += condition * lyr_top * (htop - hbot);
120
121 if (in_pos_area && ((buoy_bot >= 0) && (buoy_top < 0))) {
122 in_pos_area = false;
123 const float log_pbot = std::log10(pbot);
124 const float log_ptop = std::log10(ptop);
125 const float buoy_diff = buoy_top - buoy_bot;
126 const float log_pres_diff = log_ptop - log_pbot;
127 const float log_eql_pres =
128 log_pbot - buoy_bot * (log_pres_diff / buoy_diff);
129 eql_pres = std::max(std::pow(10.0f, log_eql_pres), pres_arr[N - 1]);
130 if (pos_buoy > pos_buoy_max) {
131 pos_buoy_max = pos_buoy;
132 lfc_pres_final = lfc_pres;
133 eql_pres_final = eql_pres;
134 }
135 }
136 pbot = ptop;
137 hbot = htop;
138 buoy_bot = buoy_top;
139 }
140 if (in_pos_area) {
141 eql_pres = pres_arr[N - 1];
142 if (pos_buoy > pos_buoy_max) {
143 pos_buoy_max = pos_buoy;
144 lfc_pres_final = lfc_pres;
145 eql_pres_final = eql_pres;
146 }
147 }
148 if (pos_buoy_max > 0.0f) {
149 this->lfc_pressure = lfc_pres_final;
150 this->eql_pressure = eql_pres_final;
151 }
152}

154void Parcel::cape_cinh(const float pres_arr[], const float hght_arr[],
155 const float buoy_arr[], const ptrdiff_t N) {
156 if (this->lcl_pressure == MISSING) return;
157 find_lfc_el(pres_arr, hght_arr, buoy_arr, N);
158 if ((this->lfc_pressure != MISSING) && (this->eql_pressure != MISSING)) {
159 PressureLayer lfc_el = {this->lfc_pressure, this->eql_pressure};
160 PressureLayer lpl_lfc = {this->pres, this->lfc_pressure};
161 HeightLayer lfc_el_ht =
162 pressure_layer_to_height(lfc_el, pres_arr, hght_arr, N);
163 HeightLayer lpl_lfc_ht =
164 pressure_layer_to_height(lpl_lfc, pres_arr, hght_arr, N);
165
166 this->cinh =
167 integrate_layer_trapz(lpl_lfc_ht, buoy_arr, hght_arr, N, -1);
168 this->cape = integrate_layer_trapz(lfc_el_ht, buoy_arr, hght_arr, N, 1);
169 }
