27float wobf(float temperature) {
28#ifndef NO_QC
29 if (temperature == MISSING) return MISSING;
30#endif
31 float pol;
32 const float x = temperature - ZEROCNK - 20.0f;
33 if (x <= 0.0f) {
34 pol = 1.0f +
35 x * (-8.841660499999999e-03f +
36 x * (1.4714143e-04f +
37 x * (-9.671989000000001e-07f +
38 x * (-3.2607217e-08f + x * (-3.8598073e-10f)))));
39 pol = pol * pol;
40 return (15.13f / (pol * pol)) + ZEROCNK;
41 } else {
42 pol = x * (4.9618922e-07f +
43 x * (-6.1059365e-09f +
44 x * (3.9401551e-11f +
45 x * (-1.2588129e-13f + x * (1.6688280e-16f)))));
46 pol = 1.0f + x * (3.6182989e-03f + x * (-1.3603273e-05f + pol));
47 pol = pol * pol;
48 return (29.93f / (pol * pol) + 0.96f * x - 14.8f) + ZEROCNK;
49 }
50}

181float saturated_lift(float pressure, float theta_sat, float converge) {
182#ifndef NO_QC
183 if ((pressure == MISSING) || (theta_sat == MISSING)) {
184 return MISSING;
185 }
186#endif
187
188 if (std::fabs(pressure - THETA_REF_PRESSURE) <= converge) return theta_sat;
189
190 const float pwrp = std::pow(pressure / THETA_REF_PRESSURE, ROCP);
191 // get the temperature
192 float t1 = theta_sat * pwrp;
193 float e1 = wobf(t1) - wobf(theta_sat);
194 float rate = 1.0f;
195 float eor = 999;
196 float t2;
197 int condition = 0;
198 // Testing the original showed that only
199 // 5 or so iterations are needed, but
200 // double that just in case. It'll exit
201 // early if it converges anyway.
202 for (int iter = 0; iter < 10; ++iter) {
203 t2 = t1 - e1 * rate;
204 float e2 = (t2) / pwrp;
205 e2 = e2 + wobf(t2) - wobf(e2) - theta_sat;
206
207 eor = e2 * rate;
208 rate = (t2 - t1) / (e2 - e1);
209 t1 = t2;
210 e1 = e2;
211 condition |= (std::fabs(eor) <= converge);
212 if (condition) break;
213 }
214 return t2 - eor;
215}

217float wetlift(float pressure, float temperature, float lifted_pressure) {
218#ifndef NO_QC
219 if ((temperature == MISSING) || (pressure == MISSING) ||

221 return MISSING;

227 // some Wobus voodo

229 const float wott = wobf(temperature);

237float _solve_cm1(float& pcl_pres_next, float& pcl_pi_next, float& pcl_t_next,
238 float& pcl_rv_next, float& pcl_rl_next, float& pcl_ri_next,
239 float pcl_pres_prev, float pcl_t_prev, float pcl_theta_prev,
240 float pcl_rv_prev, float pcl_rl_prev, float pcl_ri_prev,
241 float rv_total, bool ascending, bool ice = false,
242 float converge = 0.0002f) {
243 // first guess - use the theta of the parcel
244 // before lifting, and update the first guess
245 // accordingly
246 float pcl_theta_last = pcl_theta_prev;
247 float pcl_theta_next = pcl_theta_prev;
248 bool not_converged = true;
249
250 while (not_converged) {
251 pcl_t_next = pcl_theta_last * pcl_pi_next;
252
253 float fliq = 1.0;
254 float fice = 0.0;
255 if (ice) {
256 fliq = std::max(
257 std::min((pcl_t_next - 233.15f) / (ZEROCNK - 233.15f), 1.0f),
258 0.0f);
259 fice = 1.0f - fliq;
260 }
261
262 const float rv_term = fliq * mixratio(pcl_pres_next, pcl_t_next) +
263 fice * mixratio_ice(pcl_pres_next, pcl_t_next);
264 if (ascending) {
265 pcl_rv_next = std::min(rv_total, rv_term);
266 pcl_ri_next = std::max(fice * (rv_total - pcl_rv_next), 0.0f);
267 pcl_rl_next = std::max(rv_total - pcl_rv_next - pcl_ri_next, 0.0f);
268 } else {
269 pcl_rv_next = rv_term;
270 pcl_rl_next = fliq * (rv_total - pcl_rv_next);
271 pcl_ri_next = fice * (rv_total - pcl_rv_next);
272 }
273
274 const float tbar = 0.5f * (pcl_t_prev + pcl_t_next);
275 const float rvbar = 0.5f * (pcl_rv_prev + pcl_rv_next);
276 const float rlbar = 0.5f * (pcl_rl_prev + pcl_rl_next);
277 const float ribar = 0.5f * (pcl_ri_prev + pcl_ri_next);
278
279 const float LHV = LV1 - LV2 * tbar;
280 const float LHS = LS1 - LS2 * tbar;
281
282 const float RM = RDGAS + RVGAS * rvbar;
283 const float CPM =
284 CP_DRYAIR + CP_VAPOR * rvbar + CP_LIQUID * rlbar + CP_ICE * ribar;
285 const float term =
286 LHV * (pcl_rl_next - pcl_rl_prev) / (CPM * tbar) +
287 LHS * (pcl_ri_next - pcl_ri_prev) / (CPM * tbar) +
288 (RM / CPM - ROCP) * std::log(pcl_pres_next / pcl_pres_prev);
289
290 pcl_theta_next = pcl_theta_prev * std::exp(term);
291
292 if (std::abs(pcl_theta_next - pcl_theta_last) > converge) {
293 pcl_theta_last =
294 pcl_theta_last + 0.3 * (pcl_theta_next - pcl_theta_last);
295 } else {
296 not_converged = false;
297 }
298 }
299 return pcl_theta_next;
300}

302float moist_adiabat_cm1(float pressure, float temperature, float new_pressure,
303 float& rv_total, float& rv, float& rl, float& ri,
304 float pres_incr, float converge, adiabat ma_type) {
305#ifndef NO_QC
306 if ((temperature == MISSING) || (pressure == MISSING) ||
307 (new_pressure == MISSING)) {
308 return MISSING;
309 }
310#endif
311
312 // set up solver variables
313 const bool ice = (ma_type >= adiabat::pseudo_ice) ? true : false;
314 float dp = new_pressure - pressure;
315 const bool ascending = std::signbit(dp);
316 const int n_iters =
317 (std::abs(dp) < pres_incr) ? 1 : 1 + (int)(std::abs(dp) / pres_incr);
318 dp = (n_iters == 1) ? dp : dp / (float)n_iters;
319
320 // Start by setting the "top" variables.
321 float pcl_theta_next = theta(pressure, temperature);
322 float pcl_pres_next = pressure;
323 float pcl_pi_next = std::pow(pressure / THETA_REF_PRESSURE, ROCP);
324 float pcl_t_next = pcl_theta_next * pcl_pi_next;
325 float pcl_rv_next = rv;
326 float pcl_rl_next = rl;
327 float pcl_ri_next = ri;
328
329 // Iterate the required number of times to reach the new pressure
330 // level from the old one in increments of dp
331 for (int i = 0; i < n_iters; ++i) {
332 float pcl_theta_prev = pcl_theta_next;
333 float pcl_pres_prev = pcl_pres_next;
334 float pcl_rv_prev = pcl_rv_next;
335 float pcl_rl_prev = pcl_rl_next;
336 float pcl_ri_prev = pcl_ri_next;
337 float pcl_t_prev = pcl_t_next;
338
339 pcl_pres_next += dp;
340 pcl_pi_next = std::pow(pcl_pres_next / THETA_REF_PRESSURE, ROCP);
341
342 // call the iterative solver to get the new parcel
343 // theta accounting for liquid (and ice if enabled)
344 pcl_theta_next = _solve_cm1(
345 pcl_pres_next, pcl_pi_next, pcl_t_next, pcl_rv_next, pcl_rl_next,
346 pcl_ri_next, pcl_pres_prev, pcl_t_prev, pcl_theta_prev, pcl_rv_prev,
347 pcl_rl_prev, pcl_ri_prev, rv_total, ascending, ice, converge);
348 /*printf("%f %f\t%f\t%f %f\n", pcl_pres_prev, pcl_theta_prev,*/
349 /* pcl_pres_next, pcl_rv_next, rv_total);*/
350
351 if ((ma_type == adiabat::pseudo_liq) ||
352 (ma_type == adiabat::pseudo_ice)) {
353 rv_total = pcl_rv_next;
354 pcl_rl_next = 0.0f;
355 pcl_ri_next = 0.0f;
356 }
357 }
358 pcl_t_next = pcl_theta_next * pcl_pi_next;
359 rv = pcl_rv_next;
360 rl = pcl_rl_next;
361 ri = pcl_ri_next;
362 return pcl_t_next;
363}

424float lapse_rate(HeightLayer layer_agl, const float height[],
425 const float temperature[], const std::ptrdiff_t N) {
426#ifndef NO_QC
427 if ((layer_agl.bottom == MISSING) || (layer_agl.top == MISSING)) {
428 return MISSING;
429 }
430#endif
431
432 // convert from agl to msl
433 layer_agl.bottom += height[0];
434 layer_agl.top += height[0];
435
436 // bounds check the height layer
437 if (layer_agl.bottom < height[0]) {
438 layer_agl.bottom = height[0];
439 }
440 if (layer_agl.top > height[N - 1]) {
441 layer_agl.top = height[N - 1];
442 }
443
444 // lower and upper temperature
445 const float tmpc_l =
446 interp_height(layer_agl.bottom, height, temperature, N);
447 const float tmpc_u = interp_height(layer_agl.top, height, temperature, N);
448#ifndef NO_QC
449 if ((tmpc_l == MISSING) || (tmpc_u == MISSING)) {
450 return MISSING;
451 }
452#endif
453
454 // dT/dz, positive (definition of lapse rate), in km
455 const float dz = layer_agl.top - layer_agl.bottom;
456 return ((tmpc_u - tmpc_l) / dz) * -1000.0f;
457}

483 const float height[], const float temperature[],
