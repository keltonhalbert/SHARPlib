27float wobf(const float temperature) {
28#ifndef NO_QC
29 if (temperature == MISSING) return MISSING;
30#endif
31 float pol;
32 const float x = temperature - ZEROCNK - 20.0f;
33 if (x <= 0.0f) {
34 pol = 1.0f +
35 x * (-8.841660499999999e-03f +
36 x * (1.4714143e-04f +
37 x * (-9.671989000000001e-07f +
38 x * (-3.2607217e-08f + x * (-3.8598073e-10f)))));
39 pol = pol * pol;
40 return (15.13f / (pol * pol)) + ZEROCNK;
41 } else {
42 pol = x * (4.9618922e-07f +
43 x * (-6.1059365e-09f +
44 x * (3.9401551e-11f +
45 x * (-1.2588129e-13f + x * (1.6688280e-16f)))));
46 pol = 1.0f + x * (3.6182989e-03f + x * (-1.3603273e-05f + pol));
47 pol = pol * pol;
48 return (29.93f / (pol * pol) + 0.96f * x - 14.8f) + ZEROCNK;
49 }
50}

171float saturated_lift(float pressure, float theta_sat, const float converge) {
172#ifndef NO_QC
173 if ((pressure == MISSING) || (theta_sat == MISSING)) {
174 return MISSING;
175 }
176#endif
177
178 if (std::fabs(pressure - THETA_REF_PRESSURE) <= converge) return theta_sat;
179
180 const float pwrp = std::pow(pressure / THETA_REF_PRESSURE, ROCP);
181 // get the temperature
182 float t1 = theta_sat * pwrp;
183 float e1 = wobf(t1) - wobf(theta_sat);
184 float rate = 1.0f;
185 float eor = 999;
186 float t2;
187 int condition = 0;
188 // Testing the original showed that only
189 // 5 or so iterations are needed, but
190 // double that just in case. It'll exit
191 // early if it converges anyway.
192 for (int iter = 0; iter < 10; ++iter) {
193 t2 = t1 - e1 * rate;
194 float e2 = (t2) / pwrp;
195 e2 = e2 + wobf(t2) - wobf(e2) - theta_sat;
196
197 eor = e2 * rate;
198 rate = (t2 - t1) / (e2 - e1);
199 t1 = t2;
200 e1 = e2;
201 condition |= (std::fabs(eor) <= converge);
202 if (condition) break;
203 }
204 return t2 - eor;
205}

207float wetlift(float pressure, float temperature, float lifted_pressure) {
208#ifndef NO_QC
209 if ((temperature == MISSING) || (pressure == MISSING) ||
210 (lifted_pressure == MISSING)) {
211 return MISSING;
212 }
213#endif
214
215 // parcels potential temperature
216 const float pcl_theta = theta(pressure, temperature, THETA_REF_PRESSURE);
217 // some Wobus voodo
218 const float woth = wobf(pcl_theta);
219 const float wott = wobf(temperature);
220 // This is the wet bulb potential temperature
221 const float pcl_thetaw = pcl_theta - woth + wott;
222 // get the temperature that crosses the moist adiabat at
223 // this pressure level
224 return saturated_lift(lifted_pressure, pcl_thetaw);

227float _solve_cm1(float& pcl_pres_next, float& pcl_pi_next, float& pcl_t_next,
228 float& pcl_rv_next, float& pcl_rl_next, float& pcl_ri_next,
229 const float pcl_pres_prev, const float pcl_t_prev,
230 const float pcl_theta_prev, const float pcl_rv_prev,
231 const float pcl_rl_prev, const float pcl_ri_prev,
232 const float rv_total, const bool ascending,
233 const bool ice = false, const float converge = 0.0002f) {
234 // first guess - use the theta of the parcel
235 // before lifting, and update the first guess
236 // accordingly
237 float pcl_theta_last = pcl_theta_prev;
238 float pcl_theta_next = pcl_theta_prev;
239 bool not_converged = true;
240
241 while (not_converged) {
242 pcl_t_next = pcl_theta_last * pcl_pi_next;
243
244 float fliq = 1.0;
245 float fice = 0.0;
246 if (ice) {
247 fliq = std::max(
248 std::min((pcl_t_next - 233.15f) / (ZEROCNK - 233.15f), 1.0f),
249 0.0f);
250 fice = 1.0f - fliq;
251 }
252
253 const float rv_term = fliq * mixratio(pcl_pres_next, pcl_t_next) +
254 fice * mixratio_ice(pcl_pres_next, pcl_t_next);
255 if (ascending) {
256 pcl_rv_next = std::min(rv_total, rv_term);
257 pcl_ri_next = std::max(fice * (rv_total - pcl_rv_next), 0.0f);
258 pcl_rl_next = std::max(rv_total - pcl_rv_next - pcl_ri_next, 0.0f);
259 } else {
260 pcl_rv_next = rv_term;
261 pcl_rl_next = fliq * (rv_total - pcl_rv_next);
262 pcl_ri_next = fice * (rv_total - pcl_rv_next);
263 }
264
265 const float tbar = 0.5f * (pcl_t_prev + pcl_t_next);
266 const float rvbar = 0.5f * (pcl_rv_prev + pcl_rv_next);
267 const float rlbar = 0.5f * (pcl_rl_prev + pcl_rl_next);
268 const float ribar = 0.5f * (pcl_ri_prev + pcl_ri_next);
269
270 const float LHV = LV1 - LV2 * tbar;
271 const float LHS = LS1 - LS2 * tbar;
272
273 const float RM = RDGAS + RVGAS * rvbar;
274 const float CPM =
275 CP_DRYAIR + CP_VAPOR * rvbar + CP_LIQUID * rlbar + CP_ICE * ribar;
276 const float term =
277 LHV * (pcl_rl_next - pcl_rl_prev) / (CPM * tbar) +
278 LHS * (pcl_ri_next - pcl_ri_prev) / (CPM * tbar) +
279 (RM / CPM - ROCP) * std::log(pcl_pres_next / pcl_pres_prev);
280
281 pcl_theta_next = pcl_theta_prev * std::exp(term);
282
283 if (std::abs(pcl_theta_next - pcl_theta_last) > converge) {
284 pcl_theta_last =
285 pcl_theta_last + 0.3 * (pcl_theta_next - pcl_theta_last);
286 } else {
287 not_converged = false;
288 }
289 }
290 return pcl_theta_next;
291}

293float moist_adiabat_cm1(float pressure, float temperature, float new_pressure,
294 float& rv_total, float& rv, float& rl, float& ri,
295 const float pres_incr, const float converge,
296 const adiabat ma_type) {
297 // set up solver variables
298 const bool ice = (ma_type >= adiabat::pseudo_ice) ? true : false;
299 float dp = new_pressure - pressure;
300 const bool ascending = std::signbit(dp);
301 const int n_iters =
302 (std::abs(dp) < pres_incr) ? 1 : 1 + (int)(std::abs(dp) / pres_incr);
303 dp = (n_iters == 1) ? dp : dp / (float)n_iters;
304
305 // Start by setting the "top" variables.
306 float pcl_theta_next = theta(pressure, temperature);
307 float pcl_pres_next = pressure;
308 float pcl_pi_next = std::pow(pressure / THETA_REF_PRESSURE, ROCP);
309 float pcl_t_next = pcl_theta_next * pcl_pi_next;
310 float pcl_rv_next = rv;
311 float pcl_rl_next = rl;
312 float pcl_ri_next = ri;
313
314 // Iterate the required number of times to reach the new pressure
315 // level from the old one in increments of dp
316 for (int i = 0; i < n_iters; ++i) {
317 float pcl_theta_prev = pcl_theta_next;
318 float pcl_pres_prev = pcl_pres_next;
319 float pcl_rv_prev = pcl_rv_next;
320 float pcl_rl_prev = pcl_rl_next;
321 float pcl_ri_prev = pcl_ri_next;
322 float pcl_t_prev = pcl_t_next;
323
324 pcl_pres_next += dp;
325 pcl_pi_next = std::pow(pcl_pres_next / THETA_REF_PRESSURE, ROCP);
326
327 // call the iterative solver to get the new parcel
328 // theta accounting for liquid (and ice if enabled)
329 pcl_theta_next = _solve_cm1(
330 pcl_pres_next, pcl_pi_next, pcl_t_next, pcl_rv_next, pcl_rl_next,
331 pcl_ri_next, pcl_pres_prev, pcl_t_prev, pcl_theta_prev, pcl_rv_prev,
332 pcl_rl_prev, pcl_ri_prev, rv_total, ascending, ice, converge);
333 /*printf("%f %f\t%f\t%f %f\n", pcl_pres_prev, pcl_theta_prev,*/
334 /* pcl_pres_next, pcl_rv_next, rv_total);*/
335
336 if ((ma_type == adiabat::pseudo_liq) ||
337 (ma_type == adiabat::pseudo_ice)) {
338 rv_total = pcl_rv_next;
339 pcl_rl_next = 0.0f;
340 pcl_ri_next = 0.0f;
341 }
342 }
343 pcl_t_next = pcl_theta_next * pcl_pi_next;
344 rv = pcl_rv_next;
345 rl = pcl_rl_next;
346 ri = pcl_ri_next;
347 return pcl_t_next;
348}

409float lapse_rate(HeightLayer layer_agl, const float height[],
410 const float temperature[], const std::ptrdiff_t N) {
411#ifndef NO_QC
412 if ((layer_agl.bottom == MISSING) || (layer_agl.top == MISSING)) {
413 return MISSING;
414 }
415#endif
416
417 // convert from agl to msl
418 layer_agl.bottom += height[0];
419 layer_agl.top += height[0];
420
421 // bounds check the height layer
422 if (layer_agl.bottom < height[0]) {
423 layer_agl.bottom = height[0];
424 }
425 if (layer_agl.top > height[N - 1]) {
426 layer_agl.top = height[N - 1];
427 }
428
429 // lower and upper temperature
430 const float tmpc_l =
431 interp_height(layer_agl.bottom, height, temperature, N);
432 const float tmpc_u = interp_height(layer_agl.top, height, temperature, N);
433#ifndef NO_QC
434 if ((tmpc_l == MISSING) || (tmpc_u == MISSING)) {
435 return MISSING;
436 }
437#endif
438
439 // dT/dz, positive (definition of lapse rate), in km
440 const float dz = layer_agl.top - layer_agl.bottom;
441 return ((tmpc_u - tmpc_l) / dz) * -1000.0f;
442}

467float lapse_rate_max(HeightLayer layer_agl, const float depth,
468 const float height[], const float temperature[],
469 const std::ptrdiff_t N, HeightLayer* max_lyr) {

473 HeightLayer lyr = {z, z + depth};

476 max_lr = lr;
477 if (max_lyr) {
478 max_lyr->bottom = lyr.bottom;
