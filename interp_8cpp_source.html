24float interp_height(const float height_val, const float height_arr[],
25 const float data_arr[], const std::ptrdiff_t N) {
26#ifndef NO_QC
27 if (height_val == MISSING) return MISSING;
28 if (std::isnan(height_val)) return MISSING;
29 // If the height value is beyond the top of the profile,
30 // or below the surface, we can't reasonably extrapolate
31 if ((height_val > height_arr[N - 1]) || (height_val < height_arr[0]))
32 return MISSING;
33#endif
34
35 static constexpr auto comp = std::less<float>();
36 std::ptrdiff_t idx_top = upper_bound(height_arr, N, height_val, comp);
37 std::ptrdiff_t idx_bot = idx_top - 1;
38
39#ifndef NO_QC
40 for (; idx_bot > 0; --idx_bot) {
41 if (data_arr[idx_bot] != MISSING) break;
42 }
43
44 for (; idx_top < N; ++idx_top) {
45 if (data_arr[idx_top] != MISSING) break;
46 }
47
48 // in the case the data are still missing at this point,
49 // return a missing value
50 if ((data_arr[idx_bot] == MISSING) || (data_arr[idx_top] == MISSING))
51 return MISSING;
52#endif
53
54 const float height_bot = height_arr[idx_bot];
55 const float height_top = height_arr[idx_top];
56 const float data_bot = data_arr[idx_bot];
57 const float data_top = data_arr[idx_top];
58
59 // normalize the distance between values
60 // to a range of 0-1 for the lerp routine
61 const float dz_norm = (height_val - height_bot) / (height_top - height_bot);
62
63 // return the linear interpolation
64 return lerp(data_bot, data_top, dz_norm);
65}

115float find_first_pressure(const float data_val, const float *pressure_arr,
116 const float *data_arr, const std::ptrdiff_t N) {
117#ifndef NO_QC
118 if (data_val == MISSING) {
119 return MISSING;
120 }
121#endif
122
123 float val0, val1;
124 for (std::ptrdiff_t k = 1; k < N; ++k) {
125 val0 = data_arr[k - 1];
126 val1 = data_arr[k];
127
128 if (val0 == data_val) {
129 return pressure_arr[k - 1];
130 }
131 if (val1 == data_val) {
132 return pressure_arr[k];
133 }
134
135 // data are decreasing
136 if ((val0 > data_val) && (val1 < data_val)) {
137 const float logp_bot = std::log10(pressure_arr[k - 1]);
138 const float logp_top = std::log10(pressure_arr[k]);
139
140 const float d_norm = (val0 - data_val) / (val0 - val1);
141
142 // return the linear interpolation
143 return std::pow(10, lerp(logp_bot, logp_top, d_norm));
144 }
145
146 // data are increasing
147 if ((val0 < data_val) && (val1 > data_val)) {
148 const float logp_bot = std::log10(pressure_arr[k - 1]);
149 const float logp_top = std::log10(pressure_arr[k]);
150
151 const float d_norm = (data_val - val0) / (val1 - val0);
152
153 // return the linear interpolation
154 return std::pow(10, lerp(logp_bot, logp_top, d_norm));
155 }
156 }
157
158 return MISSING;
159}
