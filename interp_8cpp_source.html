24float interp_height(const float height_val, const float height_arr[],
25 const float data_arr[], const std::ptrdiff_t N) {
26#ifndef NO_QC
27 if (height_val == MISSING) return MISSING;
28 if (std::isnan(height_val)) return MISSING;
29 // If the height value is beyond the top of the profile,
30 // or below the surface, we can't reasonably extrapolate
31 if ((height_val > height_arr[N - 1]) || (height_val < height_arr[0]))
32 return MISSING;
33#endif
34
35 static constexpr auto comp = std::less<float>();
36 std::ptrdiff_t idx_top = upper_bound(height_arr, N, height_val, comp);
37 std::ptrdiff_t idx_bot = idx_top - 1;
38
39#ifndef NO_QC
40 for (; idx_bot > 0; --idx_bot) {
41 if (data_arr[idx_bot] != MISSING) break;
42 }
43
44 for (; idx_top < N; ++idx_top) {
45 if (data_arr[idx_top] != MISSING) break;
46 }
47
48 // in the case the data are still missing at this point,
49 // return a missing value
50 if ((data_arr[idx_bot] == MISSING) || (data_arr[idx_top] == MISSING))
51 return MISSING;
52#endif
53
54 const float height_bot = height_arr[idx_bot];
55 const float height_top = height_arr[idx_top];
56 const float data_bot = data_arr[idx_bot];
57 const float data_top = data_arr[idx_top];
58
59 // normalize the distance between values
60 // to a range of 0-1 for the lerp routine
61 const float dz_norm = (height_val - height_bot) / (height_top - height_bot);
62
63 // return the linear interpolation
64 return lerp(data_bot, data_top, dz_norm);
65}

67float interp_pressure(const float pressure_val, const float pressure_arr[],
68 const float data_arr[], const std::ptrdiff_t N) {

70 if (pressure_val == MISSING) return MISSING;

73 // or below the surface, we can't reasonably extrapolate

96 return MISSING;
97#endif
98

100 const float pressure_top = pressure_arr[idx_top];
101 const float data_bot = data_arr[idx_bot];
102 const float data_top = data_arr[idx_top];
103
104 // In order to linearly interpolate pressure properly, distance needs
105 // to be calculated in log10(pressure) coordinates and normalized
106 // between 0 and 1 for the lerp routine.

108 (std::log10(pressure_bot) - std::log10(pressure_val)) /
109 (std::log10(pressure_bot) - std::log10(pressure_top));
110
111 // return the linear interpolation
112 return lerp(data_bot, data_top, dp_norm);
113}

115float find_first_pressure(const float data_val, const float pressure_arr[],
116 const float data_arr[], const std::ptrdiff_t N) {
117 std::ptrdiff_t k_start = 0;
118#ifndef NO_QC
119 if (data_val == MISSING) {
120 return MISSING;
121 }
122 for (; k_start < N; ++k_start) {
123 if (data_arr[k_start] != MISSING) break;
124 }
125#endif
126
127 for (std::ptrdiff_t k = k_start + 1; k < N; ++k) {
128 float val0 = data_arr[k_start];
129 float val1 = data_arr[k];
130#ifndef NO_QC
131 if (val1 == MISSING) continue;
132#endif
133 if (val0 == data_val) return pressure_arr[k_start];
134 if (val1 == data_val) return pressure_arr[k];
135
136 if ((data_val - val0) * (data_val - val1) < 0) {
137 const float logp_bot = std::log10(pressure_arr[k_start]);
138 const float logp_top = std::log10(pressure_arr[k]);
139
140 const float d_norm = (data_val - val0) / (val1 - val0);
141
142 return std::pow(10, lerp(logp_bot, logp_top, d_norm));
143 }
144
145 k_start = k;
146 }
147
148 return MISSING;
149}

151float find_first_height(const float data_val, const float height_arr[],
152 const float data_arr[], const std::ptrdiff_t N) {
153 std::ptrdiff_t k_start = 0;
154#ifndef NO_QC
155 if (data_val == MISSING) {
156 return MISSING;
157 }
158 for (; k_start < N; ++k_start) {
159 if (data_arr[k_start] != MISSING) break;
160 }
161#endif
162
163 for (std::ptrdiff_t k = 1; k < N; ++k) {
164 float val0 = data_arr[k_start];
165 float val1 = data_arr[k];
166#ifndef NO_QC
167 if (val1 == MISSING) continue;
168#endif
169 if (val0 == data_val) return height_arr[k_start];
170 if (val1 == data_val) return height_arr[k];
171
172 // will have a negative sign if levels straddle
173 // the point being searched for
174 if ((data_val - val0) * (data_val - val1) < 0) {
175 const float hght_bot = height_arr[k_start];
176 const float hght_top = height_arr[k];
177
178 const float d_norm = (data_val - val0) / (val1 - val0);
179
180 return lerp(hght_bot, hght_top, d_norm);
181 }
182
183 k_start = k;
184 }
185
186 return MISSING;
187}
