27float interp_height(float height_val, const float height_arr[],
28 const float data_arr[], const std::ptrdiff_t N) {
29#ifndef NO_QC
30 if (height_val == MISSING) return MISSING;
31 if (std::isnan(height_val)) return MISSING;
32 // If the height value is beyond the top of the profile,
33 // or below the surface, we can't reasonably extrapolate
34 if ((height_val > height_arr[N - 1]) || (height_val < height_arr[0]))
35 return MISSING;
36#endif
37
38 static constexpr auto comp = std::less<float>();
39 std::ptrdiff_t idx_top = upper_bound(height_arr, N, height_val, comp);
40 std::ptrdiff_t idx_bot = idx_top - 1;
41
42#ifndef NO_QC
43 for (; idx_bot > 0; --idx_bot) {
44 if (data_arr[idx_bot] != MISSING) break;
45 }
46
47 for (; idx_top < N; ++idx_top) {
48 if (data_arr[idx_top] != MISSING) break;
49 }
50
51 // in the case the data are still missing at this point,
52 // return a missing value
53 if ((data_arr[idx_bot] == MISSING) || (data_arr[idx_top] == MISSING))
54 return MISSING;
55#endif
56
57 const float height_bot = height_arr[idx_bot];
58 const float height_top = height_arr[idx_top];
59 const float data_bot = data_arr[idx_bot];
60 const float data_top = data_arr[idx_top];
61
62 // normalize the distance between values
63 // to a range of 0-1 for the lerp routine
64 const float dz_norm = (height_val - height_bot) / (height_top - height_bot);
65
66 // return the linear interpolation
67 return lerp(data_bot, data_top, dz_norm);
68}

70float interp_pressure(float pressure_val, const float pressure_arr[],
71 const float data_arr[], const std::ptrdiff_t N) {
72#ifndef NO_QC
73 if (pressure_val == MISSING) return MISSING;
74 if (std::isnan(pressure_val)) return MISSING;
75 // If the pressure value is beyond the top of the profile,
76 // or below the surface, we can't reasonably extrapolate

78 (pressure_val > pressure_arr[0])) {
79 return MISSING;
80 }
81#endif

94 }

97 // return a missing value
98 if ((data_arr[idx_bot] == MISSING) || (data_arr[idx_top] == MISSING))
99 return MISSING;
100#endif
101
102 const float pressure_bot = pressure_arr[idx_bot];
103 const float pressure_top = pressure_arr[idx_top];
104 const float data_bot = data_arr[idx_bot];
105 const float data_top = data_arr[idx_top];
106
107 // In order to linearly interpolate pressure properly, distance needs
108 // to be calculated in log10(pressure) coordinates and normalized
109 // between 0 and 1 for the lerp routine.
110 const float dp_norm =
111 (std::log10(pressure_bot) - std::log10(pressure_val)) /
112 (std::log10(pressure_bot) - std::log10(pressure_top));
113
114 // return the linear interpolation
115 return lerp(data_bot, data_top, dp_norm);
116}
